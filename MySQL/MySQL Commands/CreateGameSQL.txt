CREATE TABLE Game (
	match_id int NOT NULL AUTO_INCREMENT,
    white_id int NOT NULL REFERENCES Account(id),
    black_id int NOT NULL REFERENCES Account(id),
    turn_num int,
    white_in BOOLEAN,
    black_in BOOLEAN, 
    white_username varchar(15),
    black_username varchar(15),
	PRIMARY KEY(match_id)
);
